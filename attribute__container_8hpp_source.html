123 template <typename T>
124 requires(!StringLike<T>)
125 Attribute<std::decay_t<T>> *add(const std::string &name, T &&value)
126 {
127 using ValueType = std::decay_t<T>;
128
129 auto attr = std::make_unique<Attribute<ValueType>>(name,

131
132 auto *ptr = attr.get();

134 auto [it, inserted] = attributes_.try_emplace(name, std::move(attr));
135
136 if (!inserted)
137 throw std::runtime_error("Attribute already exists: " + name);
138
139 // keep track of insertion order
140 insertion_order_.push_back(name);

244 template <typename T> T *try_value(const std::string &name)

275 template <typename T> T &value(const std::string &name)
276 {
277 if (auto *ptr = try_value<T>(name)) return *ptr;

279 auto *attr = find(name);
